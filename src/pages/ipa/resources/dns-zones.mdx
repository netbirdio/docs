## Create a DNS Zone  {{ tag: 'POST' , label: '/api/dns/zones' }}

<Row>
  <Col>
    Creates a new custom DNS zone
            
    ### Request-Body Parameters
    
    <Properties><Property name="name" type="string" required={true} minLen={1} maxLen={255}>
        
            Zone name identifier
        
        </Property>
    <Property name="domain" type="string" required={true}>
        
            Zone domain (FQDN)
        
        </Property>
    <Property name="enabled" type="boolean" required={false}>
        
            Zone status
        
        </Property>
    <Property name="enable_search_domain" type="boolean" required={true}>
        
            Enable this zone as a search domain
        
        </Property>
    <Property name="distribution_groups" type="string[]" required={true}>
        
            Group IDs that defines groups of peers that will resolve this zone
        
        </Property>
    </Properties>

    
       </Col>

  <Col sticky>

  "name": "Office Zone",
